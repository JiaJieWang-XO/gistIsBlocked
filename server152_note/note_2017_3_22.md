# 关于在Shell脚本中判别服务器
```shell
set -e
set -u
hostName=$(hostname)
echo "hostname is $hostName"
if [ $hostName = "DB14" ] ; then
  echo "Do something on DB14"
elif [ $hostName = "DB15" ] ; then
  echo "Do something on DB15"
else
  echo "Do something else on $hostName"
fi

```
